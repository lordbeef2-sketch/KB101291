````text
#!/bin/bash
#=========================================
# Wrapper to simplify snapshot management
#=========================================
#
#  Please set the DATA_DIR and BACKUP_DIR locations to point to the locations in your deployment
#
#  If running under Cygwin, please make sure you use the notation /cygrive/X/path
#     where X is the drive letter and path is the path under that drive
cd "$(dirname ${BASH_SOURCE[0]})"
DATA_DIR=/data/
BACKUP_DIR=/backups
mkdir -p $BACKUP_DIR
rm -fr $BACKUP_DIR/*.tar
./backup.sh -dir $DATA_DIR -rf $BACKUP_DIR


````
