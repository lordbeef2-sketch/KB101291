#!/bin/bash
die() {
  local _ret=$1
  test -n "$_ret" || _ret=1
  test "$_PRINT_HELP" = yes && print_help >&2
  echo "$1" >&2
  exit ${_ret}
}

begins_with_short_option() {
  local first_option all_short_options
  all_short_options='h'
  first_option="${1:0:1}"
  test "$all_short_options" = "${all_short_options/$first_option/}" && return 1 || return 0
}

# THE DEFAULTS INITIALIZATION - OPTIONALS
DATABASE_DIR=$cassandra_database
RECOVERY_FILE_DIR=$rf
export JAVA_TOOL_OPTIONS=-Dcom.sun.jndi.rmiURLParsing=legacy

_arg_dir=$DATABASE_DIR
_arg_rf=$RECOVERY_FILE

print_help() {
  printf '%s\n' ""
  printf 'Usage: %s [--dir <arg>] [--rf <arg>] [-h|--help]\n' "$0"
  printf '\t%s\n' "--dir: cassandra database path (default: DATABASE_DIR)"
  printf '\t%s\n' "--rf: full path to snapshot archive (default: RECOVERY_FILE)"
  printf '\t%s\n' "-h,--help: Prints help"
}

parse_commandline() {
  while test $# -gt 0; do
    _key="$1"
    case "$_key" in
    --dir | -dir)
      test $# -lt 2 && die "Missing value for the optional argument '$_key'." 1
      _arg_dir="$2"
      DATABASE_DIR=$_arg_dir
      shift
      ;;
    --dir=*)
      _arg_dir="${_key##--dir=}"
      DATABASE_DIR=$_arg_dir
      ;;
    --rf | -rf)
      test $# -lt 2 && die "Missing value for the optional argument '$_key'." 1
      _arg_rf="$2"
      RECOVERY_FILE_DIR=$_arg_rf
      shift
      ;;
    --rf=*)
      _arg_rf="${_key##--rf=}"
      RECOVERY_FILE_DIR=$_arg_rf
      ;;
    -h | --help)
      print_help
      exit 0
      ;;
    -h*)
      print_help
      exit 0
      ;;
    *)
      _PRINT_HELP=yes die "FATAL ERROR: Got an unexpected argument '$1'" 1
      ;;
    esac
    shift
  done
}

parse_commandline "$@"

osname=$(uname -o)

echo "DB=$DATABASE_DIR"
echo "RF=$RECOVERY_FILE_DIR"

###  If using non-default credentials change user_name and password to reflect your installation
user_name=cassandra
password=cassandra

#FILE_NAME="cassandra_backup_$(date "+%Y.%m.%d-%H.%M.%S").tar"
FILE_NAME="$(hostname)_$(date "+%Y.%m.%d-%H.%M.%S").tar"

init() {
  init_database_dir
  init_recovery_file_dir
}
mkdir -p $RECOVERY_FILE_DIR
echo "$(date) - Starting backup" >>$RECOVERY_FILE_DIR/backup.txt

flush_data_to_disk() {
  echo "$(date) - Stopping cassandra compactions if they were running" >>$RECOVERY_FILE_DIR/backup.txt
  if [ "$osname" = "Cygwin" ]; then
    nodetool.bat -h localhost -u $user_name -pw $password -p 7199 stop
  else
    nodetool -h localhost -u $user_name -pw $password -p 7199 stop
  fi
  echo "$(date) - Flushing data to disk" >>$RECOVERY_FILE_DIR/backup.txt
  if [ "$osname" = "Cygwin" ]; then
    nodetool.bat -h localhost -u $user_name -pw $password -p 7199 flush
    check_for_errors
  else
    nodetool -h localhost -u $user_name -pw $password -p 7199 flush
    check_for_errors
  fi
}

await_flush() {
  sleep 10
}

snapshot_cassandra_database() {
  echo "$(date) - Taking a snapshot of all key spaces" >>$RECOVERY_FILE_DIR/backup.txt
  if [ "$osname" = "Cygwin" ]; then
    nodetool.bat -h localhost -u $user_name -pw $password -p 7199 snapshot
    check_for_errors
  else
    nodetool -h localhost -u $user_name -pw $password -p 7199 snapshot
    check_for_errors
  fi
}

init_database_dir() {
  if [ ! -d "$DATABASE_DIR/data/" ]; then
    request_db_dir
  fi
}

init_recovery_file_dir() {
  if [ ! -d "$RECOVERY_FILE_DIR" ]; then
    request_recovery_file_dir
  fi
}

request_recovery_file_dir() {
  need_dir=true
  while $need_dir; do
    read -p "Please enter backup file location. I.e. /home/user/backups/: " RECOVERY_FILE_DIR
    if [ ! -d "$RECOVERY_FILE_DIR" ]; then
      echo "No backup directory found $RECOVERY_FILE_DIR"
      echo "Wrong dir entered."
    else
      need_dir=false
    fi
  done
}

request_db_dir() {
  need_dir=true
  while $need_dir; do
    read -p "Please enter Cassandra database directory. I.e. /var/lib/cassandra: " DATABASE_DIR
    if [ ! -d "$DATABASE_DIR/data/" ]; then
      echo "No directory $DATABASE_DIR/data/"
      echo "Wrong directory entered."
    else
      need_dir=false
    fi
  done
}

archive_snapshots_for_backup() {
  echo "$(date) - Archiving snapshots for backup" >>$RECOVERY_FILE_DIR/backup.txt
  (
    cd "$DATABASE_DIR/data"
    find . -type f -path "*/snapshots/*/*" | tar -cpf $RECOVERY_FILE_DIR/$FILE_NAME -T - --hard-dereference
    #mv $FILE_NAME $RECOVERY_FILE_DIR
  )
}

remove_snapshot() {
  echo "$(date) - Removing old snapshots" >>$RECOVERY_FILE_DIR/backup.txt
  if [ "$osname" = "Cygwin" ]; then
    nodetool.bat -h localhost -u $user_name -pw $password -p 7199 clearsnapshot --all
    check_for_errors
  else
    nodetool -h localhost -u $user_name -pw $password -p 7199 clearsnapshot --all
    check_for_errors
  fi
}

check_for_errors() {
  rc=$?
  if [[ $rc != 0 ]]; then
    exit $rc
  fi
}

finish() {
  echo "$(date) - Backup file created successfully. Location: $RECOVERY_FILE_DIR/$FILE_NAME" >>$RECOVERY_FILE_DIR/backup.txt
}

fail() {
  echo "$(date)Failed to backup cassandra, check console output for more details" >>$RECOVERY_FILE_DIR/backup.txt
}

{
  init && flush_data_to_disk && await_flush && snapshot_cassandra_database && archive_snapshots_for_backup && remove_snapshot && finish

} || {
  fail
}
